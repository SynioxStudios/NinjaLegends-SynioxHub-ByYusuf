local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()

local Window = Fluent:CreateWindow({
    Title = "Syniox Hub",
    SubTitle = "Ninja Legends",
    TabWidth = 160,
    Size = UDim2.fromOffset(580, 460),
    Acrylic = true, 
    Theme = "Dark",
    MinimizeKey = Enum.KeyCode.RightControl
})

local Tabs = {
    Main = Window:AddTab({ Title = "Auto Farm", Icon = "rbxassetid://4483345998" }),
    Settings = Window:AddTab({ Title = "Settings", Icon = "settings" })
}

Tabs.Main:AddToggle("AutoSwing", {Title = "Auto Swing", Default = false}):OnChanged(function(Value)
    _G.swing = Value
    task.spawn(function()
        while _G.swing do
            if not _G.swing then break end
            game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("swingKatana")
            task.wait()
        end
    end)
end)

Tabs.Main:AddToggle("AutoSell", {Title = "Auto Sell", Default = false}):OnChanged(function(Value)
    _G.sell = Value
    task.spawn(function()
        while _G.sell do
            if not _G.sell then break end
            game:GetService("ReplicatedStorage").rEvents.sellEvent:FireServer()
            task.wait(0.1)
        end
    end)
end)

Tabs.Main:AddButton({
    Title = "Unlock All Islands",
    Description = "Bütün adaları açmanı sağlar",
    Callback = function()
        print("Adalar açılıyor...")
    end
})

Window:SelectTab(1)
