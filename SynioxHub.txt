-- SYNIOX HUB - DIRECT LOADER
-- Küçük buton vs. her şeyi sildik, direkt senin GUI'ni yüklüyoruz.
local SynioxLib = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/Fluent/refs/heads/main/Fluent.txt"))()

-- 1. ANA PENCERE VE SEKME AYARLARI
-- (Bu kısımlar kütüphanenin çalışması için gereken temel tanımlardır)
local Window = SynioxLib:CreateWindow({
    Title = "Syniox Hub",
    SubTitle = "Muscle Legends",
    TabWidth = 160,
    Size = UDim2.fromOffset(580, 460),
    Acrylic = true,
    Theme = "Dark",
    MinimizeKey = Enum.KeyCode.LeftControl -- Menüyü gizlemek/açmak için CTRL
})

local Tabs = {
    Main = Window:AddTab({ Title = "Main", Icon = "home" }),
    Settings = Window:AddTab({ Title = "Settings", Icon = "settings" })
}

-- 2. ÖRNEK FONKSİYONLAR
-- Görseldeki gibi "Main" sekmesine özellikleri ekliyoruz
Tabs.Main:AddToggle("AutoLift", {Title = "Auto Lift", Default = false})
Tabs.Main:AddToggle("FastTools", {Title = "Fast Tools", Default = false})
