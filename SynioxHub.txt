local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()

local Window = Fluent:CreateWindow({
    Title = "Syniox Hub",
    SubTitle = "Ninja Legends",
    TabWidth = 160,
    Size = UDim2.fromOffset(450, 300),
    Acrylic = false, 
    Theme = "Dark",
    MinimizeKey = Enum.KeyCode.RightControl
})

local ToggleButton = Instance.new("TextButton")
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Parent = game:GetService("CoreGui")
ToggleButton.Parent = ScreenGui
ToggleButton.Size = UDim2.new(0, 50, 0, 50)
ToggleButton.Position = UDim2.new(0.1, 0, 0.1, 0)
ToggleButton.Text = "AÃ§"
ToggleButton.BackgroundColor3 = Color3.fromRGB(41, 74, 122)
ToggleButton.TextColor3 = Color3.fromRGB(255, 255, 255)
Instance.new("UICorner", ToggleButton).CornerRadius = UDim.new(0, 8)
ToggleButton.Draggable = true

ToggleButton.MouseButton1Click:Connect(function()
    game:GetService("VirtualInputManager"):SendKeyEvent(true, Enum.KeyCode.RightControl, false, game)
end)

local Tabs = {
    Main = Window:AddTab({ Title = "Auto Farm", Icon = "rbxassetid://4483345998" }),
    Settings = Window:AddTab({ Title = "Settings", Icon = "settings" })
}

Tabs.Main:AddToggle("AutoSwing", {Title = "Auto Swing", Default = false}):OnChanged(function(Value)
    _G.swing = Value
    task.spawn(function()
        while _G.swing do
            if not _G.swing then break end
            game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("swingKatana")
            task.wait()
        end
    end)
end)

Tabs.Main:AddToggle("AutoSell", {Title = "Auto Sell", Default = false}):OnChanged(function(Value)
    _G.sell = Value
    task.spawn(function()
        while _G.sell do
            if not _G.sell then break end
            game:GetService("ReplicatedStorage").rEvents.sellEvent:FireServer()
            task.wait(0.1)
        end
    end)
end)

Window:SelectTab(1)
