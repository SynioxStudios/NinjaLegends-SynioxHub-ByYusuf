local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/ByYusufGui-Syniox/refs/heads/main/SynioxGui.txt", true))()
local displayName = game.Players.LocalPlayer.DisplayName

local window = library:AddWindow("Syniox Public | Ninja Legends || HELLO - ".. displayName, {
    title_bar = {
        Color3.fromRGB(160, 0, 0), 
        Color3.fromRGB(40, 0, 0),  
        Color3.fromRGB(0, 0, 0)    
    }, 
    title_bar_transparency = 0.1, 
    background = {
        Color3.fromRGB(0, 0, 0), 
        Color3.fromRGB(10, 10, 10), 
        Color3.fromRGB(20, 20, 20)
    }, 
    background_transparency = 0.1, 
    main_color = Color3.fromRGB(160, 0, 0), 
    min_size = Vector2.new(450, 450), 
    can_resize = true 
})

local FarmTab = window:AddTab("Farm")
FarmTab:AddSwitch("Auto Swing", function(state)
    _G.swing = state
    task.spawn(function()
        while _G.swing do
            game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("swingKatana")
            task.wait()
        end
    end)
end)
FarmTab:AddSwitch("Auto Sell", function(state)
    _G.sell = state
    task.spawn(function()
        while _G.sell do
            if game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                game.workspace.sellAreaCircles.sellAreaCircle15.circleInner.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
                task.wait(0.1)
                game.workspace.sellAreaCircles.sellAreaCircle15.circleInner.CFrame = game.Workspace.Part.CFrame
            end
            task.wait(0.1)
        end
    end)
end)
FarmTab:AddSwitch("Auto Collect Chi", function(state)
    _G.chi = state
    task.spawn(function()
        while _G.chi do
            for _,v in pairs(game.Workspace.chiItems:GetChildren()) do
                if v:IsA("Part") or v:IsA("MeshPart") then
                    v.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
                end
            end
            task.wait(0.5)
        end
    end)
end)

local CombatTab = window:AddTab("Combat")
CombatTab:AddSwitch("Auto Boss (Robot)", function(state)
    _G.boss = state
    task.spawn(function()
        while _G.boss do
            if game.Workspace:FindFirstChild("RobotBoss") then
                game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = game.Workspace.RobotBoss.HumanoidRootPart.CFrame * CFrame.new(0,0,5)
                game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("swingKatana")
            end
            task.wait()
        end
    end)
end)

            end

local IslandTab = window:AddTab("Islands")
IslandTab:AddButton("Unlock All Islands", function()
    local islands = {
        CFrame.new(80, 766, -188), CFrame.new(-128, 39439, 173), CFrame.new(183, 46010, 36),
        CFrame.new(188, 59594, 24), CFrame.new(166, 52607, 34), CFrame.new(180, 33206, 28),
        CFrame.new(172, 80498, 26), CFrame.new(175, 92955, 30), CFrame.new(168, 115200, 25)
    }
    for _, cf in ipairs(islands) do
        game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = cf
        task.wait(0.5)
    end
end)
