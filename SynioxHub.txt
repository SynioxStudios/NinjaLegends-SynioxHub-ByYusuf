local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/ByYusufGui-Syniox/refs/heads/main/SynioxGui.txt"))()

local Window = Library:AddWindow("Syniox Hub | Ninja Legends", {
    main_color = Color3.fromRGB(41, 74, 122),
    min_size = Vector2.new(450, 400),
    toggle_key = Enum.KeyCode.RightControl,
    can_resize = true
})

local MainTab = Window:AddTab("Auto Farm")
local TeleportTab = Window:AddTab("Islands")
local MeditateTab = Window:AddTab("Meditate")
local BuyTab = Window:AddTab("Auto Buy")
local MiscTab = Window:AddTab("Misc")

MainTab:AddToggle("Auto Swing", function(state)
    _G.swing = state
    task.spawn(function()
        while _G.swing do
            game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("swingKatana")
            task.wait()
        end
    end)
end)

MainTab:AddToggle("Auto Sell", function(state)
    _G.sell = state
    task.spawn(function()
        while _G.sell do
            if game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                game.workspace.sellAreaCircles.sellAreaCircle15.circleInner.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
                task.wait(0.1)
                game.workspace.sellAreaCircles.sellAreaCircle15.circleInner.CFrame = game.Workspace.Part.CFrame
            end
            task.wait(0.1)
        end
    end)
end)

MainTab:AddToggle("Auto Collect Chi", function(state)
    _G.chi = state
    task.spawn(function()
        while _G.chi do
            for _,v in pairs(game.Workspace.chiItems:GetChildren()) do
                if v:IsA("Part") or v:IsA("MeshPart") then
                    v.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
                end
            end
            task.wait(0.5)
        end
    end)
end)

TeleportTab:AddButton("Unlock All Islands", function()
    local islands = {
        CFrame.new(80, 766, -188), CFrame.new(-128, 39439, 173), CFrame.new(183, 46010, 36),
        CFrame.new(188, 59594, 24), CFrame.new(166, 52607, 34), CFrame.new(180, 33206, 28),
        CFrame.new(172, 80498, 26), CFrame.new(175, 92955, 30), CFrame.new(168, 115200, 25)
    }
    for _, cf in ipairs(islands) do
        game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = cf
        task.wait(0.3)
    end
end)

MeditateTab:AddButton("Mystical Waters", function()
    game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(320, 30383, -12)
end)

MeditateTab:AddButton("Midnight Waters", function()
    game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(315, 30383, -20)
end)

BuyTab:AddToggle("Auto Buy Swords", function(state)
    _G.swords = state
    task.spawn(function()
        while _G.swords do
            game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("buyAllSwords", "Blazing Vortex Island")
            task.wait(0.5)
        end
    end)
end)

BuyTab:AddToggle("Auto Buy Belts", function(state)
    _G.belts = state
    task.spawn(function()
        while _G.belts do
            game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("buyAllBelts", "Blazing Vortex Island")
            task.wait(0.5)
        end
    end)
end)

BuyTab:AddToggle("Auto Buy Skills", function(state)
    _G.skills = state
    task.spawn(function()
        while _G.skills do
            game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("buyAllSkills", "Blazing Vortex Island")
            task.wait(0.5)
        end
    end)
end)

MiscTab:AddToggle("Auto Rebirth", function(state)
    _G.rebirth = state
    task.spawn(function()
        while _G.rebirth do
            game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("checkRebirth")
            task.wait(1)
        end
    end)
end)

MiscTab:AddButton("No Fog & Lag Fix", function()
    game:GetService("Lighting").FogEnd = 999999
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("ParticleEmitter") or v:IsA("Trail") then v.Enabled = false end
    end
end)
