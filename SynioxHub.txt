local Fluent = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/Fluent/refs/heads/main/Fluent.txt"))()

local Window = Fluent:CreateWindow({
    Title = "Syniox Hub",
    SubTitle = "Ninja Legends",
    TabWidth = 160,
    Size = UDim2.fromOffset(550, 380),
    Acrylic = true,
    Theme = "Dark",
    MinimizeKey = Enum.KeyCode.LeftControl
})

local Tabs = {
    Main = Window:AddTab({ Title = "Auto Farm", Icon = "swords" }),
    Shop = Window:AddTab({ Title = "Auto Buy", Icon = "shopping-cart" }),
    Misc = Window:AddTab({ Title = "Misc", Icon = "zap" })
}

Tabs.Main:AddToggle("AutoSwing", {Title = "Auto Swing", Default = false}):OnChanged(function(Value)
    _G.swing = Value
    while _G.swing do
        task.wait()
        game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("swingKatana")
    end
end)

Tabs.Main:AddToggle("AutoSell", {Title = "Auto Sell", Default = false}):OnChanged(function(Value)
    _G.sell = Value
    while _G.sell do
        task.wait()
        game.workspace.sellAreaCircles.sellAreaCircle15.circleInner.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
        task.wait()
        game.workspace.sellAreaCircles.sellAreaCircle15.circleInner.CFrame = game.Workspace.Part.CFrame
    end
end)

Tabs.Shop:AddButton({
    Title = "Buy All Items (Blazing Vortex)",
    Callback = function()
        local remote = game:GetService("Players").LocalPlayer.ninjaEvent
        remote:FireServer("buyAllSwords", "Blazing Vortex Island")
        remote:FireServer("buyAllBelts", "Blazing Vortex Island")
        remote:FireServer("buyAllSkills", "Blazing Vortex Island")
    end
})

Tabs.Misc:AddToggle("Invisibility", {Title = "Invisibility", Default = false}):OnChanged(function(Value)
    _G.invis = Value
    while _G.invis do
        task.wait()
        game.Players.LocalPlayer.ninjaEvent:FireServer("goInvisible")
    end
end)

Fluent:Notify({Title = "Syniox Hub", Content = "Ninja Legends Script Ready!", Duration = 5})
