local Fluent = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/Fluent/refs/heads/main/Fluent.txt"))()

local Window = Fluent:CreateWindow({
    Title = "Syniox Hub",
    Size = UDim2.fromOffset(550, 380)
})

local FarmTab = Window:AddTab({ Title = "Auto Farm" })

FarmTab:AddToggle("Auto Swing", function(Value)
    _G.swing = Value
    while _G.swing do
        task.wait()
        game:GetService("Players").LocalPlayer.ninjaEvent:FireServer("swingKatana")
    end
end)

FarmTab:AddToggle("Auto Sell", function(Value)
    _G.sell = Value
    while _G.sell do
        task.wait()
        game.workspace.sellAreaCircles.sellAreaCircle15.circleInner.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
        task.wait(0.1)
        game.workspace.sellAreaCircles.sellAreaCircle15.circleInner.CFrame = game.Workspace.Part.CFrame
    end
end)

FarmTab:AddButton({
    Title = "Unlock All Islands",
    Callback = function()
        local islands = {
            CFrame.new(80, 766, -188),
            CFrame.new(-128, 39439, 173),
            CFrame.new(183, 46010, 36)
        }
        for _, cf in ipairs(islands) do
            game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = cf
            task.wait(0.2)
        end
    end
})
